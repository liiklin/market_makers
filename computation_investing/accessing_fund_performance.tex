    \centerline{\Large Metrics for Assessing Fund Performance}
    \section{Return}
    \begin{equation}
    return = \frac{price[end] }{price[begin]}- 1
    \end{equation}
    \section{Standard Deviation of Daily Return}
    
    \begin{equation}
        daily\_rets[i] = \frac{value[i]}{value[i-1]} -1 
    \end{equation} 
    
    \begin{equation}
        std\_metric = std(daily\_rets)
    \end{equation}
